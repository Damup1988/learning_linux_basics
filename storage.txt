# list block devices
lsblk
ls /dev/ -l | grep "^b"
sudo fdisk -l /dev/sda

# create a partition on a disk
gdisk /dev/sdb

# define file system
mkfs.ext4 /dev/sdb1
mkdir /mnt/ext4;
mount /dev/sdb1 /mnt/ext4

# show mounts
mount

# show file system for a disk
sudo blkid /dev/sda1

# show persistent mounts
cat /dev/fstab

# for persistent mount
echo "/dev/sdb1 /mnt/ext4 ext4 rw 0 0" >> /etc/fstab

# install LVM
apt install lvm2

# create a physical volume
pvcreate /dev/sdb

# create a volume group
vgcreate caleston_vg /dev/sdb

# list all physical volumes
pvdisplay

# create logical volumes
lvcreate -L 1G -n vol1 caleston_vg

# list logical volumes
lvdisplay
lvs

# create a file system on a logical volume
mkfs.ext4 /dev/caleston_vg/vol1
mount -t ext4 /dev/caleston_vg/vol1 /mnt/vol1

# check available space on a volume group
vgs

# increase a volume
lvresize -L +1G -n /dev/caleston_vg/vol1
df -hP /mnt/vol1
resize2fs /dev/caleston_vg/vol1
