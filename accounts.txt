# create a new user
sudo adduser user_name
sudo useradd -u 1010 -g john -s /bin/sh -m john

# create a group with id
sudo groupadd -g id_number group_name

# get info regarding accounts
cat /etc/passwd

# get info about groups
cat /etc/group

# get info about user account
id 

# show currently logged users
who

# show history of logins
last

# switch to root
su -

# get config of sudo
cat /etc/sudoers

# get stored passwords
cat /etc/shadow
